 198def polyinterp(points, x_min_bound=None, x_max_bound=None, plot=False):
 199    """
 200    Gives the minimizer and minimum of the interpolating polynomial over given points
 201    based on function and derivative information. Defaults to bisection if no critical
 202    points are valid.
 203
 204    Based on polyinterp.m Matlab function in minFunc by Mark Schmidt with some slight
 205    modifications.
 206
 207    Implemented by: Hao-Jun Michael Shi and Dheevatsa Mudigere
 208    Last edited 12/6/18.
 209
 210    Inputs:
 211        points (nparray): two-dimensional array with each point of form [x f g]
 212        x_min_bound (float): minimum value that brackets minimum (default: minimum of points)
 213        x_max_bound (float): maximum value that brackets minimum (default: maximum of points)
 214        plot (bool): plot interpolating polynomial
 215
 216    Outputs:
 217        x_sol (float): minimizer of interpolating polynomial
 218        F_min (float): minimum of interpolating polynomial
 219
 220    Note:
 221      . Set f or g to np.nan if they are unknown
 222
 223    """
 224    no_points = points.shape[0]
 225    order = np.sum(1 - np.isnan(points[:, 1:3]).astype('int')) - 1
 226
 227    x_min = np.min(points[:, 0])
 228    x_max = np.max(points[:, 0])
 229
 230    # compute bounds of interpolation area
 231    if x_min_bound is None:
 232        x_min_bound = x_min
 233    if x_max_bound is None:
 234        x_max_bound = x_max
 235
 236    # explicit formula for quadratic interpolation
 237    if no_points == 2 and order == 2 and plot is False:
 238        # Solution to quadratic interpolation is given by:
 239        # a = -(f1 - f2 - g1(x1 - x2))/(x1 - x2)^2
 240        # x_min = x1 - g1/(2a)
 241        # if x1 = 0, then is given by:
 242        # x_min = - (g1*x2^2)/(2(f2 - f1 - g1*x2))
 243
 244        if points[0, 0] == 0:
 245            x_sol = -points[0, 2] * points[1, 0] ** 2 / (
 246                        2 * (points[1, 1] - points[0, 1] - points[0, 2] * points[1, 0]))
 247        else:
 248            a = -(points[0, 1] - points[1, 1] - points[0, 2] * (points[0, 0] - points[1, 0])) / (
 249                        points[0, 0] - points[1, 0]) ** 2
 250            x_sol = points[0, 0] - points[0, 2] / (2 * a)
 251
 252        x_sol = np.minimum(np.maximum(x_min_bound, x_sol), x_max_bound)
 253
 254    # explicit formula for cubic interpolation
 255    elif no_points == 2 and order == 3 and plot is False:
 256        # Solution to cubic interpolation is given by:
 257        # d1 = g1 + g2 - 3((f1 - f2)/(x1 - x2))
 258        # d2 = sqrt(d1^2 - g1*g2)
 259        # x_min = x2 - (x2 - x1)*((g2 + d2 - d1)/(g2 - g1 + 2*d2))
 260        d1 = points[0, 2] + points[1, 2] - 3 * ((points[0, 1] - points[1, 1]) / (points[0, 0] - points[1, 0]))
 261        d2 = np.sqrt(d1 ** 2 - points[0, 2] * points[1, 2])
 262        if np.isreal(d2):
 263            x_sol = points[1, 0] - (points[1, 0] - points[0, 0]) * (
 264                        (points[1, 2] + d2 - d1) / (points[1, 2] - points[0, 2] + 2 * d2))
 265            x_sol = np.minimum(np.maximum(x_min_bound, x_sol), x_max_bound)
 266        else:
 267            x_sol = (x_max_bound + x_min_bound) / 2
 268
 269    # solve linear system
 270    else:
 271        # define linear constraints
 272        A = np.zeros((0, order + 1))
 273        b = np.zeros((0, 1))
 274
 275        # add linear constraints on function values
 276        for i in range(no_points):
 277            if not np.isnan(points[i, 1]):
 278                constraint = np.zeros((1, order + 1))
 279                for j in range(order, -1, -1):
 280                    constraint[0, order - j] = points[i, 0] ** j
 281                A = np.append(A, constraint, 0)
 282                b = np.append(b, points[i, 1])
 283
 284        # add linear constraints on gradient values
 285        for i in range(no_points):
 286            if not np.isnan(points[i, 2]):
 287                constraint = np.zeros((1, order + 1))
 288                for j in range(order):
 289                    constraint[0, j] = (order - j) * points[i, 0] ** (order - j - 1)
 290                A = np.append(A, constraint, 0)
 291                b = np.append(b, points[i, 2])
 292
 293        # check if system is solvable
 294        if A.shape[0] != A.shape[1] or np.linalg.matrix_rank(A) != A.shape[0]:
 295            x_sol = (x_min_bound + x_max_bound) / 2
 296            f_min = np.Inf
 297        else:
 298            # solve linear system for interpolating polynomial
 299            coeff = np.linalg.solve(A, b)
 300
 301            # compute critical points
 302            dcoeff = np.zeros(order)
 303            for i in range(len(coeff) - 1):
 304                dcoeff[i] = coeff[i] * (order - i)
 305
 306            crit_pts = np.array([x_min_bound, x_max_bound])
 307            crit_pts = np.append(crit_pts, points[:, 0])
 308
 309            if not np.isinf(dcoeff).any():
 310                roots = np.roots(dcoeff)
 311                crit_pts = np.append(crit_pts, roots)
 312
 313            # test critical points
 314            f_min = np.Inf
 315            x_sol = (x_min_bound + x_max_bound) / 2  # defaults to bisection
 316            for crit_pt in crit_pts:
 317                if np.isreal(crit_pt) and crit_pt >= x_min_bound and crit_pt <= x_max_bound:
 318                    F_cp = np.polyval(coeff, crit_pt)
 319                    if np.isreal(F_cp) and F_cp < f_min:
 320                        x_sol = np.real(crit_pt)
 321                        f_min = np.real(F_cp)
 322
 323            if (plot):
 324                plt.figure()
 325                x = np.arange(x_min_bound, x_max_bound, (x_max_bound - x_min_bound) / 10000)
 326                f = np.polyval(coeff, x)
 327                plt.plot(x, f)
 328                plt.plot(x_sol, f_min, 'x')
 329
 330    return x_sol

 351            Options:
 352                'None': uses steplength designated in algorithm
 353                'Armijo': uses Armijo backtracking line search
 354                'Wolfe': uses Armijo-Wolfe bracketing line search

 356        debug (bool): debugging mode

 369        Mathematics of Computation 35.151 (1980): 773-782.
 370    [6] Nocedal, Jorge, and Stephen J. Wright. "Numerical Optimization." Springer New York,
 371        2006.

 470    def two_loop_recursion(self, vec):
 471        """
 472        Performs two-loop recursion on given vector to obtain Hv.
 473
 474        Inputs:
 475            vec (tensor): 1-D tensor to apply two-loop recursion to
 476
 477        Output:
 478            r (tensor): matrix-vector product Hv
 479
 480        """
 481
 482        group = self.param_groups[0]
 483        history_size = group['history_size']
 484
 485        state = self.state['global_state']
 486        old_dirs = state.get('old_dirs')  # change in gradients
 487        old_stps = state.get('old_stps')  # change in iterates
 488        H_diag = state.get('H_diag')
 489
 490        # compute the product of the inverse Hessian approximation and the gradient
 491        num_old = len(old_dirs)
 492
 493        if 'rho' not in state:
 494            state['rho'] = [None] * history_size
 495            state['alpha'] = [None] * history_size
 496        rho = state['rho']
 497        alpha = state['alpha']
 498
 499        for i in range(num_old):
 500            rho[i] = 1. / old_stps[i].dot(old_dirs[i])
 501
 502        q = vec
 503        for i in range(num_old - 1, -1, -1):
 504            alpha[i] = old_dirs[i].dot(q) * rho[i]
 505            q.add_(old_stps[i], alpha=-alpha[i])
 506
 507        # multiply by initial Hessian
 508        # r/d is the final direction
 509        r = torch.mul(q, H_diag)
 510        for i in range(num_old):
 511            beta = old_stps[i].dot(r) * rho[i]
 512            r.add_(old_dirs[i], alpha=alpha[i] - beta)
 513
 514        return r

 516    def curvature_update(self, flat_grad, eps=1e-2, damping=False):
 517        """
 518        Performs curvature update.
 519
 520        Inputs:
 521            flat_grad (tensor): 1-D tensor of flattened gradient for computing
 522                gradient difference with previously stored gradient
 523            eps (float): constant for curvature pair rejection or damping (default: 1e-2)
 524            damping (bool): flag for using Powell damping (default: False)
 525        """
 526
 527        assert len(self.param_groups) == 1
 528
 529        # load parameters
 530        if (eps <= 0):
 531            raise (ValueError('Invalid eps; must be positive.'))
 532
 533        group = self.param_groups[0]
 534        history_size = group['history_size']
 535        debug = group['debug']
 536
 537        # variables cached in state (for tracing)
 538        state = self.state['global_state']
 539        fail = state.get('fail')
 540
 541        # check if line search failed
 542        if not fail:
 543
 544            d = state.get('d')
 545            t = state.get('t')
 546            old_dirs = state.get('old_dirs')
 547            old_stps = state.get('old_stps')
 548            H_diag = state.get('H_diag')
 549            prev_flat_grad = state.get('prev_flat_grad')
 550            Bs = state.get('Bs')
 551
 552            # compute y's
 553            y = flat_grad.sub(prev_flat_grad)
 554            s = d.mul(t)
 555            sBs = s.dot(Bs)
 556            ys = y.dot(s)  # y*s
 557
 558            # update L-BFGS matrix
 559            if ys > eps * sBs or damping == True:
 560
 561                # perform Powell damping
 562                if damping == True and ys < eps * sBs:
 563                    if debug:
 564                        print('Applying Powell damping...')
 565                    theta = ((1 - eps) * sBs) / (sBs - ys)
 566                    y = theta * y + (1 - theta) * Bs
 567
 568                # updating memory
 569                if len(old_dirs) == history_size:
 570                    # shift history by one (limited-memory)
 571                    old_dirs.pop(0)
 572                    old_stps.pop(0)
 573
 574                # store new direction/step
 575                old_dirs.append(s)
 576                old_stps.append(y)
 577
 578                # update scale of initial Hessian approximation
 579                H_diag = ys / y.dot(y)  # (y*y)
 580
 581                state['old_dirs'] = old_dirs
 582                state['old_stps'] = old_stps
 583                state['H_diag'] = H_diag
 584
 585            else:
 586                # save skip
 587                state['curv_skips'] += 1
 588                if debug:
 589                    print('Curvature pair skipped due to failed criterion')
 590
 591        else:
 592            # save skip
 593            state['fail_skips'] += 1
 594            if debug:
 595                print('Line search failed; curvature pair update skipped')
 596
 597        return

 599    def _step(self, p_k, g_Ok, g_Sk=None, options=None):

 602

 642                closure_eval (int): number of closure evaluations

 653                g_new (tensor): gradient at new iterate

 655                ls_step (int): number of backtracks

 658                desc_dir (bool): descent direction flag
 659                    True: p_k is descent direction with respect to the line search
 660                    function
 661                    False: p_k is not a descent direction with respect to the line
 662                    search function
 663                fail (bool): failure flag
 664                    True: line search reached maximum number of iterations, failed
 665                    False: line search succeeded
 666
 667        Notes:
 668          . If encountering line search failure in the deterministic setting, one
 669            should try increasing the maximum number of line search steps max_ls.
 670
 671        """
 672

 674            options = {}

 677        # load parameter options
 678        group = self.param_groups[0]

 680        line_search = group['line_search']
 681        dtype = group['dtype']
 682        debug = group['debug']

 728                    F_k = closure()
 729                    closure_eval += 1

 783            # begin print for debug mode
 784            if ls_debug:
 785                print(
 786                    '==================================== Begin Armijo line search ===================================')

 790            if gtd >= 0:

 792                if debug:
 793                    print('Not a descent direction!')

 796
 797            # store values if not in-place
 798            if not inplace:
 799                current_params = self._copy_params()
 800
 801            # update and evaluate at new point
 802            self._add_update(t, d)

 804            closure_eval += 1
 805
 806            # print info if debugging
 807            if ls_debug:
 808                print('LS Step: %d  t: %.8e  F(x+td): %.8e  F-c1*t*g*d: %.8e  F(x): %.8e'

 810

 812            while F_new > F_k + c1 * t * gtd or not is_legal(F_new):
 813

 816                    if inplace:
 817                        self._add_update(-t, d)
 818                    else:
 819                        self._load_params(current_params)
 820

 822                    F_new = closure()
 823                    closure_eval += 1

 827                else:
 828                    # store current steplength
 829                    t_new = t
 830
 831                    # compute new steplength
 832
 833                    # if first step or not interpolating, then multiply by factor
 834                    if ls_step == 0 or not interpolate or not is_legal(F_new):
 835                        t = t / eta
 836
 837                    # if second step, use function value at new point along with
 838                    # gradient and function at current iterate
 839                    elif ls_step == 1 or not is_legal(F_prev):
 840                        t = polyinterp(np.array([[0, F_k.item(), gtd.item()], [t_new, F_new.item(), np.nan]]))
 841
 842                    # otherwise, use function values at new point, previous point,
 843                    # and gradient and function at current iterate
 844                    else:
 845                        t = polyinterp(np.array([[0, F_k.item(), gtd.item()], [t_new, F_new.item(), np.nan],
 846                                                 [t_prev, F_prev.item(), np.nan]]))
 847
 848                    # if values are too extreme, adjust t
 849                    if interpolate:
 850                        if t < 1e-3 * t_new:
 851                            t = 1e-3 * t_new
 852                        elif t > 0.6 * t_new:
 853                            t = 0.6 * t_new
 854
 855                        # store old point
 856                        F_prev = F_new
 857                        t_prev = t_new
 858
 859                    # update iterate and reevaluate
 860                    if inplace:
 861                        self._add_update(t - t_new, d)
 862                    else:
 863                        self._load_params(current_params)
 864                        self._add_update(t, d)
 865
 866                    F_new = closure()
 867                    closure_eval += 1
 868                    ls_step += 1  # iterate
 869
 870                    # print info if debugging
 871                    if ls_debug:
 872                        print('LS Step: %d  t: %.8e  F(x+td):   %.8e  F-c1*t*g*d: %.8e  F(x): %.8e'
 873                              % (ls_step, t, F_new, F_k + c1 * t * gtd, F_k))
 874
 875            # store Bs
 876            if Bs is None:
 877                Bs = (g_Sk.mul(-t)).clone()
 878            else:
 879                Bs.copy_(g_Sk.mul(-t))
 880
 881            # print final steplength
 882            if ls_debug:
 883                print('Final Steplength:', t)
 884                print(
 885                    '===================================== End Armijo line search ====================================')
 886
 887            state['d'] = d
 888            state['prev_flat_grad'] = prev_flat_grad
 889            state['t'] = t
 890            state['Bs'] = Bs
 891            state['fail'] = fail
 892
 893            return F_new, t, ls_step, closure_eval, desc_dir, fail
 894
 895        # perform weak Wolfe line search
 896        elif line_search == 'Wolfe':
 897
 898            # load options
 899            if options:
 900                if 'closure' not in options.keys():
 901                    raise (ValueError('closure option not specified.'))
 902                else:
 903                    closure = options['closure']
 904
 905                if 'current_loss' not in options.keys():
 906                    F_k = closure()
 907                    closure_eval += 1
 908                else:
 909                    F_k = options['current_loss']
 910
 911                if 'gtd' not in options.keys():
 912                    gtd = g_Sk.dot(d)
 913                else:
 914                    gtd = options['gtd']
 915
 916                if 'eta' not in options.keys():
 917                    eta = 2
 918                elif options['eta'] <= 1:
 919                    raise (ValueError('Invalid eta; must be greater than 1.'))
 920                else:
 921                    eta = options['eta']
 922
 923                if 'c1' not in options.keys():
 924                    c1 = 1e-4
 925                elif options['c1'] >= 1 or options['c1'] <= 0:
 926                    raise (ValueError('Invalid c1; must be strictly between 0 and 1.'))
 927                else:
 928                    c1 = options['c1']
 929
 930                if 'c2' not in options.keys():
 931                    c2 = 0.9
 932                elif options['c2'] >= 1 or options['c2'] <= 0:
 933                    raise (ValueError('Invalid c2; must be strictly between 0 and 1.'))
 934                elif options['c2'] <= c1:
 935                    raise (ValueError('Invalid c2; must be strictly larger than c1.'))
 936                else:
 937                    c2 = options['c2']
 938
 939                if 'max_ls' not in options.keys():
 940                    max_ls = 10
 941                elif options['max_ls'] <= 0:
 942                    raise (ValueError('Invalid max_ls; must be positive.'))
 943                else:
 944                    max_ls = options['max_ls']
 945
 946                if 'interpolate' not in options.keys():
 947                    interpolate = True
 948                else:
 949                    interpolate = options['interpolate']
 950
 951                if 'inplace' not in options.keys():
 952                    inplace = True
 953                else:
 954                    inplace = options['inplace']
 955
 956                if 'ls_debug' not in options.keys():
 957                    ls_debug = False
 958                else:
 959                    ls_debug = options['ls_debug']
 960
 961            else:
 962                raise (ValueError('Options are not specified; need closure evaluating function.'))
 963            # initialize counters
 964            ls_step = 0
 965            grad_eval = 0  # tracks gradient evaluations
 966            t_prev = 0  # old steplength
 967
 968            # initialize bracketing variables and flag
 969            alpha = 0
 970            beta = float('Inf')
 971            fail = False
 972
 973            # initialize values for line search
 974            if (interpolate):
 975                F_a = F_k
 976                g_a = gtd
 977
 978                if (torch.cuda.is_available()):
 979                    F_b = torch.tensor(np.nan, dtype=dtype).cuda()
 980                    g_b = torch.tensor(np.nan, dtype=dtype).cuda()
 981                else:
 982                    F_b = torch.tensor(np.nan, dtype=dtype)
 983                    g_b = torch.tensor(np.nan, dtype=dtype)

 985            # begin print for debug mode

 988                    '==================================== Begin Wolfe line search ====================================')
 989                print('F(x): %.8e  g*d: %.8e' % (F_k, gtd))
 990
 991            # check if search direction is descent direction
 992            if gtd >= 0:
 993                desc_dir = False
 994                if debug:
 995                    print('Not a descent direction!')
 996            else:
 997                desc_dir = True
 998
 999            # store values if not in-place
1000            if not inplace:
1001                current_params = self._copy_params()
1002
1003            # update and evaluate at new point
1004            self._add_update(t, d)
1005            F_new = closure()
1006            closure_eval += 1
1007            # main loop
1008            while True:
1009
1010                # check if maximum number of line search steps have been reached
1011                if ls_step >= max_ls:
1012                    if inplace:
1013                        self._add_update(-t, d)
1014                    else:
1015                        self._load_params(current_params)

1018                    F_new = closure()

1020                    g_new = self._gather_flat_grad()
1021                    closure_eval += 1
1022                    grad_eval += 1
1023                    fail = True
1024                    break

1026                # print info if debugging
1027                if ls_debug:
1028                    print('LS Step: %d  t: %.8e  alpha: %.8e  beta: %.8e'

1031                          % (F_new, F_k + c1 * t * gtd, F_k))
1032
1033                # check Armijo condition
1034                if F_new > F_k + c1 * t * gtd:
1035
1036                    # set upper bound
1037                    beta = t
1038                    t_prev = t
1039
1040                    # update interpolation quantities
1041                    if interpolate:
1042                        F_b = F_new
1043                        if torch.cuda.is_available():
1044                            g_b = torch.tensor(np.nan, dtype=dtype).cuda()
1045                        else:
1046                            g_b = torch.tensor(np.nan, dtype=dtype)
1047
1048                else:
1049
1050                    # compute gradient
1051                    # F_new.backward()
1052                    g_new = self._gather_flat_grad()
1053                    grad_eval += 1
1054                    gtd_new = g_new.dot(d)
1055
1056                    # print info if debugging
1057                    if ls_debug:
1058                        print('Wolfe: g(x+td)*d: %.8e  c2*g*d: %.8e  gtd: %.8e'
1059                              % (gtd_new, c2 * gtd, gtd))
1060
1061                    # check curvature condition
1062                    if gtd_new < c2 * gtd:
1063
1064                        # set lower bound
1065                        alpha = t
1066                        t_prev = t
1067
1068                        # update interpolation quantities
1069                        if interpolate:
1070                            F_a = F_new
1071                            g_a = gtd_new
1072
1073                    else:
1074                        break
1075
1076                # compute new steplength
1077                # if first step or not interpolating, then bisect or multiply by factor
1078                if not interpolate or not is_legal(F_b):
1079                    if beta == float('Inf'):
1080                        t = eta * t
1081                    else:
1082                        t = (alpha + beta) / 2.0
1083
1084                # otherwise interpolate between a and b
1085                else:
1086                    t = polyinterp(np.array([[alpha, F_a.item(), g_a.item()], [beta, F_b.item(), g_b.item()]]))
1087
1088                    # if values are too extreme, adjust t
1089                    if beta == float('Inf'):
1090                        if t > 2 * eta * t_prev:
1091                            t = 2 * eta * t_prev
1092                        elif t < eta * t_prev:
1093                            t = eta * t_prev
1094                    else:
1095                        if t < alpha + 0.2 * (beta - alpha):
1096                            t = alpha + 0.2 * (beta - alpha)
1097                        elif t > (beta - alpha) / 2.0:
1098                            t = (beta - alpha) / 2.0
1099
1100                    # if we obtain nonsensical value from interpolation
1101                    if t <= 0:
1102                        t = (beta - alpha) / 2.0
1103
1104                # update parameters
1105                if inplace:
1106                    self._add_update(t - t_prev, d)
1107                else:
1108                    self._load_params(current_params)
1109                    self._add_update(t, d)
1110
1111                # evaluate closure
1112                F_new = closure()
1113                closure_eval += 1
1114                ls_step += 1
1115
1116            # store Bs
1117            if Bs is None:
1118                Bs = (g_Sk.mul(-t)).clone()
1119            else:
1120                Bs.copy_(g_Sk.mul(-t))
1121
1122            # print final steplength
1123            if ls_debug:
1124                print('Final Steplength:', t)
1125                print(
1126                    '===================================== End Wolfe line search =====================================')
1127
1128            state['d'] = d
1129            state['prev_flat_grad'] = prev_flat_grad
1130            state['t'] = t
1131            state['Bs'] = Bs
1132            state['fail'] = fail
1133
1134            return F_new, g_new, t, ls_step, closure_eval, grad_eval, desc_dir, fail
1135
1136        else:
1137
1138            # perform update
1139            self._add_update(t, d)
1140
1141            # store Bs
1142            if Bs is None:
1143                Bs = (g_Sk.mul(-t)).clone()
1144            else:
1145                Bs.copy_(g_Sk.mul(-t))
1146
1147            state['d'] = d
1148            state['prev_flat_grad'] = prev_flat_grad
1149            state['t'] = t
1150            state['Bs'] = Bs
1151            state['fail'] = False
1152
1153            return t

200    """
201    Gives the minimizer and minimum of the interpolating polynomial over given points
202    based on function and derivative information. Defaults to bisection if no critical
203    points are valid.
204
205    Based on polyinterp.m Matlab function in minFunc by Mark Schmidt with some slight
206    modifications.
207
208    Implemented by: Hao-Jun Michael Shi and Dheevatsa Mudigere
209    Last edited 12/6/18.
210
211    Inputs:
212        points (nparray): two-dimensional array with each point of form [x f g]
213        x_min_bound (float): minimum value that brackets minimum (default: minimum of points)
214        x_max_bound (float): maximum value that brackets minimum (default: maximum of points)
215        plot (bool): plot interpolating polynomial
216
217    Outputs:
218        x_sol (float): minimizer of interpolating polynomial
219        F_min (float): minimum of interpolating polynomial
220
221    Note:
222      . Set f or g to np.nan if they are unknown
223
224    """
225    no_points = points.shape[0]
226    order = np.sum(1 - np.isnan(points[:, 1:3]).astype('int')) - 1
227
228    x_min = np.min(points[:, 0])
229    x_max = np.max(points[:, 0])
230
231    # compute bounds of interpolation area
232    if x_min_bound is None:
233        x_min_bound = x_min
234    if x_max_bound is None:
235        x_max_bound = x_max
236
237    # explicit formula for quadratic interpolation
238    if no_points == 2 and order == 2 and plot is False:
239        # Solution to quadratic interpolation is given by:
240        # a = -(f1 - f2 - g1(x1 - x2))/(x1 - x2)^2
241        # x_min = x1 - g1/(2a)
242        # if x1 = 0, then is given by:
243        # x_min = - (g1*x2^2)/(2(f2 - f1 - g1*x2))
244
245        if points[0, 0] == 0:
246            x_sol = -points[0, 2] * points[1, 0] ** 2 / (
247                        2 * (points[1, 1] - points[0, 1] - points[0, 2] * points[1, 0]))
248        else:
249            a = -(points[0, 1] - points[1, 1] - points[0, 2] * (points[0, 0] - points[1, 0])) / (
250                        points[0, 0] - points[1, 0]) ** 2
251            x_sol = points[0, 0] - points[0, 2] / (2 * a)
252
253        x_sol = np.minimum(np.maximum(x_min_bound, x_sol), x_max_bound)
254
255    # explicit formula for cubic interpolation
256    elif no_points == 2 and order == 3 and plot is False:
257        # Solution to cubic interpolation is given by:
258        # d1 = g1 + g2 - 3((f1 - f2)/(x1 - x2))
259        # d2 = sqrt(d1^2 - g1*g2)
260        # x_min = x2 - (x2 - x1)*((g2 + d2 - d1)/(g2 - g1 + 2*d2))
261        d1 = points[0, 2] + points[1, 2] - 3 * ((points[0, 1] - points[1, 1]) / (points[0, 0] - points[1, 0]))
262        d2 = np.sqrt(d1 ** 2 - points[0, 2] * points[1, 2])
263        if np.isreal(d2):
264            x_sol = points[1, 0] - (points[1, 0] - points[0, 0]) * (
265                        (points[1, 2] + d2 - d1) / (points[1, 2] - points[0, 2] + 2 * d2))
266            x_sol = np.minimum(np.maximum(x_min_bound, x_sol), x_max_bound)
267        else:
268            x_sol = (x_max_bound + x_min_bound) / 2
269
270    # solve linear system
271    else:
272        # define linear constraints
273        A = np.zeros((0, order + 1))
274        b = np.zeros((0, 1))
275
276        # add linear constraints on function values
277        for i in range(no_points):
278            if not np.isnan(points[i, 1]):
279                constraint = np.zeros((1, order + 1))
280                for j in range(order, -1, -1):
281                    constraint[0, order - j] = points[i, 0] ** j
282                A = np.append(A, constraint, 0)
283                b = np.append(b, points[i, 1])
284
285        # add linear constraints on gradient values
286        for i in range(no_points):
287            if not np.isnan(points[i, 2]):
288                constraint = np.zeros((1, order + 1))
289                for j in range(order):
290                    constraint[0, j] = (order - j) * points[i, 0] ** (order - j - 1)
291                A = np.append(A, constraint, 0)
292                b = np.append(b, points[i, 2])
293
294        # check if system is solvable
295        if A.shape[0] != A.shape[1] or np.linalg.matrix_rank(A) != A.shape[0]:
296            x_sol = (x_min_bound + x_max_bound) / 2
297            f_min = np.Inf
298        else:
299            # solve linear system for interpolating polynomial
300            coeff = np.linalg.solve(A, b)
301
302            # compute critical points
303            dcoeff = np.zeros(order)
304            for i in range(len(coeff) - 1):
305                dcoeff[i] = coeff[i] * (order - i)
306
307            crit_pts = np.array([x_min_bound, x_max_bound])
308            crit_pts = np.append(crit_pts, points[:, 0])
309
310            if not np.isinf(dcoeff).any():
311                roots = np.roots(dcoeff)
312                crit_pts = np.append(crit_pts, roots)
313
314            # test critical points
315            f_min = np.Inf
316            x_sol = (x_min_bound + x_max_bound) / 2  # defaults to bisection
317            for crit_pt in crit_pts:

 471    def two_loop_recursion(self, vec):
 472        """
 473        Performs two-loop recursion on given vector to obtain Hv.
 474
 475        Inputs:
 476            vec (tensor): 1-D tensor to apply two-loop recursion to
 477
 478        Output:
 479            r (tensor): matrix-vector product Hv
 480

 482
 483        group = self.param_groups[0]
 484        history_size = group['history_size']
 485
 486        state = self.state['global_state']
 487        old_dirs = state.get('old_dirs')  # change in gradients
 488        old_stps = state.get('old_stps')  # change in iterates
 489        H_diag = state.get('H_diag')
 490
 491        # compute the product of the inverse Hessian approximation and the gradient
 492        num_old = len(old_dirs)
 493
 494        if 'rho' not in state:
 495            state['rho'] = [None] * history_size
 496            state['alpha'] = [None] * history_size
 497        rho = state['rho']
 498        alpha = state['alpha']
 499
 500        for i in range(num_old):
 501            rho[i] = 1. / old_stps[i].dot(old_dirs[i])
 502
 503        q = vec
 504        for i in range(num_old - 1, -1, -1):
 505            alpha[i] = old_dirs[i].dot(q) * rho[i]
 506            q.add_(old_stps[i], alpha=-alpha[i])
 507
 508        # multiply by initial Hessian
 509        # r/d is the final direction
 510        r = torch.mul(q, H_diag)
 511        for i in range(num_old):
 512            beta = old_stps[i].dot(r) * rho[i]
 513            r.add_(old_dirs[i], alpha=alpha[i] - beta)
 514
 515        return r

 517    def curvature_update(self, flat_grad, eps=1e-2, damping=False):
 518        """
 519        Performs curvature update.
 520
 521        Inputs:
 522            flat_grad (tensor): 1-D tensor of flattened gradient for computing
 523                gradient difference with previously stored gradient
 524            eps (float): constant for curvature pair rejection or damping (default: 1e-2)
 525            damping (bool): flag for using Powell damping (default: False)
 526        """
 527
 528        assert len(self.param_groups) == 1
 529
 530        # load parameters
 531        if (eps <= 0):
 532            raise (ValueError('Invalid eps; must be positive.'))
 533
 534        group = self.param_groups[0]
 535        history_size = group['history_size']
 536        debug = group['debug']
 537
 538        # variables cached in state (for tracing)
 539        state = self.state['global_state']
 540        fail = state.get('fail')
 541
 542        # check if line search failed
 543        if not fail:
 544
 545            d = state.get('d')
 546            t = state.get('t')
 547            old_dirs = state.get('old_dirs')
 548            old_stps = state.get('old_stps')
 549            H_diag = state.get('H_diag')
 550            prev_flat_grad = state.get('prev_flat_grad')
 551            Bs = state.get('Bs')
 552
 553            # compute y's
 554            y = flat_grad.sub(prev_flat_grad)
 555            s = d.mul(t)
 556            sBs = s.dot(Bs)
 557            ys = y.dot(s)  # y*s
 558
 559            # update L-BFGS matrix
 560            if ys > eps * sBs or damping == True:
 561
 562                # perform Powell damping
 563                if damping == True and ys < eps * sBs:
 564                    if debug:
 565                        print('Applying Powell damping...')
 566                    theta = ((1 - eps) * sBs) / (sBs - ys)
 567                    y = theta * y + (1 - theta) * Bs
 568
 569                # updating memory
 570                if len(old_dirs) == history_size:
 571                    # shift history by one (limited-memory)
 572                    old_dirs.pop(0)
 573                    old_stps.pop(0)
 574
 575                # store new direction/step
 576                old_dirs.append(s)
 577                old_stps.append(y)
 578
 579                # update scale of initial Hessian approximation
 580                H_diag = ys / y.dot(y)  # (y*y)
 581
 582                state['old_dirs'] = old_dirs
 583                state['old_stps'] = old_stps
 584                state['H_diag'] = H_diag
 585
 586            else:
 587                # save skip
 588                state['curv_skips'] += 1
 589                if debug:
 590                    print('Curvature pair skipped due to failed criterion')
 591
 592        else:
 593            # save skip
 594            state['fail_skips'] += 1

 596                print('Line search failed; curvature pair update skipped')
